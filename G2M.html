KM Travelled¶

  • H0: KM Travelled = 22.56
  • H1: KM Travelled != 22.56
In [122]:
KM = np.random.choice(final_df["KM Travelled"], sample)
tset, pval = ttest_1samp(KM, 22.56)
print("p-values",pval)
if pval < 0.05:  
    print(" we are rejecting null hypothesis")
else:
    print("we are accepting null hypothesis")
